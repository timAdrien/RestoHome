def self.included(base)
  base.class_eval do
    remove_possible_method :to_time

    def to_time
      if preserve_timezone
        @_to_time_with_instance_offset ||= getlocal(utc_offset)
      else
        @_to_time_with_system_offset ||= getlocal
      end
    end
  end
end
